\begin{definition}
An integer $n$ is \emph{even} if $n = 2k$ for some integer $k$.
\end{definition}

\begin{theorem}
The sum of two even integers is even.
\end{theorem}

\begin{proof}
Let $a$ and $b$ be even, so $a = 2m$ and $b = 2n$ for integers $m$ and
$n$. Then
\[
    a + b = 2m + 2n = 2(m + n),
\]
which is twice an integer, and therefore even. The \texttt{proof}
environment ends with an automatic $\square$.
\end{proof}
